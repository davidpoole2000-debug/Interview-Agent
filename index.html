<script>

async function runAssessment() {

  const output = document.getElementById("output");

  output.textContent = "Working...";


  let transcript;

  try {

    transcript = JSON.parse(document.getElementById("transcript").value || "[]");

  } catch (e) {

    output.textContent = "Transcript JSON is invalid.";

    return;

  }


  const body = {

    candidateName: document.getElementById("name").value || "Candidate",

    transcript,

    exerciseAnswer: document.getElementById("exercise").value || ""

  };


  const res = await fetch("/api/assess", {

    method: "POST",

    headers: { "Content-Type": "application/json" },

    body: JSON.stringify(body)

  });


  const json = await res.json();

  output.textContent = JSON.stringify(json, null, 2);

}

</script>
